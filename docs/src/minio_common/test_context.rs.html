25#[derive(Clone)]
26pub struct TestContext {
27    pub client: MinioClient,
28    pub base_url: BaseUrl,
29    pub access_key: String,
30    pub secret_key: String,
31    pub ignore_cert_check: Option<bool>,
32    pub ssl_cert_file: Option<PathBuf>,

35impl TestContext {
36    pub fn new_from_env() -> Self {
37        let run_on_ci: bool = std::env::var("CI")
38            .unwrap_or("false".into())
39            .parse()
40            .unwrap_or(false);
41        if run_on_ci {
42            let host = std::env::var("SERVER_ENDPOINT").unwrap();
43            let access_key = std::env::var("ACCESS_KEY").unwrap();
44            let secret_key = std::env::var("SECRET_KEY").unwrap();
45            let secure = std::env::var("ENABLE_HTTPS").is_ok();
46            let value = std::env::var("MINIO_SSL_CERT_FILE").unwrap();
47            let mut ssl_cert_file = None;
48            if !value.is_empty() {
49                ssl_cert_file = Some(Path::new(&value));
50            }
51            let ignore_cert_check = std::env::var("IGNORE_CERT_CHECK").is_ok();
52            let region = std::env::var("SERVER_REGION").ok();
53
54            let mut base_url: BaseUrl = host.parse().unwrap();
55            base_url.https = secure;
56            if let Some(v) = region {
57                base_url.region = Region::try_from(v.as_str()).unwrap();
58            }
59
60            let static_provider = StaticProvider::new(&access_key, &secret_key, None);
61            let client = MinioClient::new(
62                base_url.clone(),
63                Some(static_provider),
64                ssl_cert_file,
65                Some(ignore_cert_check),
66            )
67            .unwrap();
68
69            Self {
70                client,
71                base_url,
72                access_key,
73                secret_key,
74                ignore_cert_check: Some(ignore_cert_check),
75                ssl_cert_file: ssl_cert_file.map(PathBuf::from),
76            }
77        } else {
78            const DEFAULT_SERVER_ENDPOINT: &str = "http://localhost:9000/";
79            const DEFAULT_ACCESS_KEY: &str = "minioadmin";
80            const DEFAULT_SECRET_KEY: &str = "minioadmin";
81            const DEFAULT_ENABLE_HTTPS: &str = "false";
82            const DEFAULT_SSL_CERT_FILE: &str = "./tests/public.crt";
83            const DEFAULT_IGNORE_CERT_CHECK: &str = "true";
84            const DEFAULT_SERVER_REGION: &str = "us-east-1";
85
86            let host: String =
87                std::env::var("SERVER_ENDPOINT").unwrap_or(DEFAULT_SERVER_ENDPOINT.to_string());
88            log::debug!("SERVER_ENDPOINT={host}");
89            let access_key: String =
90                std::env::var("ACCESS_KEY").unwrap_or(DEFAULT_ACCESS_KEY.to_string());
91            log::debug!("ACCESS_KEY={access_key}");
92            let secret_key: String =
93                std::env::var("SECRET_KEY").unwrap_or(DEFAULT_SECRET_KEY.to_string());
94            log::debug!("SECRET_KEY=*****");
95            let secure: bool = std::env::var("ENABLE_HTTPS")
96                .unwrap_or(DEFAULT_ENABLE_HTTPS.to_string())
97                .parse()
98                .unwrap_or(false);
99            log::debug!("ENABLE_HTTPS={secure}");
100            let ssl_cert: String =
101                std::env::var("MINIO_SSL_CERT_FILE").unwrap_or(DEFAULT_SSL_CERT_FILE.to_string());
102            log::debug!("MINIO_SSL_CERT_FILE={ssl_cert}");
103            let ssl_cert_file: PathBuf = ssl_cert.into();
104            let ignore_cert_check: bool = std::env::var("IGNORE_CERT_CHECK")
105                .unwrap_or(DEFAULT_IGNORE_CERT_CHECK.to_string())
106                .parse()
107                .unwrap_or(true);
108            log::debug!("IGNORE_CERT_CHECK={ignore_cert_check}");
109            let region_str: String =
110                std::env::var("SERVER_REGION").unwrap_or(DEFAULT_SERVER_REGION.to_string());
111            log::debug!("SERVER_REGION={region_str:?}");
112
113            let mut base_url: BaseUrl = host.parse().unwrap();
114            base_url.https = secure;
115            base_url.region = Region::try_from(region_str.as_str()).unwrap();
116
117            let static_provider = StaticProvider::new(&access_key, &secret_key, None);
118            let client = MinioClient::new(
119                base_url.clone(),
120                Some(static_provider),
121                Some(&*ssl_cert_file),
122                Some(ignore_cert_check),
123            )
124            .unwrap();
125
126            Self {
127                client,
128                base_url,
129                access_key,
130                secret_key,
131                ignore_cert_check: Some(ignore_cert_check),
132                ssl_cert_file: Some(ssl_cert_file),
133            }
134        }
135    }

137    /// Creates a temporary bucket with an automatic cleanup guard.
138    ///
139    /// This function creates a new bucket and returns both its name and a `CleanupGuard`
140    /// that ensures the bucket is deleted when it goes out of scope.  
141    ///
142    /// # Returns
143    /// A tuple containing:
144    /// - `String` - The name of the created bucket.
145    /// - `CleanupGuard` - A guard that automatically deletes the bucket when dropped.
146    ///
147    /// # Example
148    /// ```no_run
149    /// let (bucket, guard) = client.create_bucket_helper().await;
150    /// println!("Created temporary bucket: {}", bucket);
151    /// // The bucket will be removed when `guard` is dropped.
152    /// ```
153    pub async fn create_bucket_helper(&self) -> (BucketName, CleanupGuard) {
154        let bucket = rand_bucket_name();
155        let _resp = self
156            .client
157            .create_bucket(&bucket)
158            .unwrap()
159            .build()
160            .send()
161            .await
162            .unwrap();
163        let guard = CleanupGuard::new(self.client.clone(), bucket.clone());
164        (bucket, guard)
165    }
