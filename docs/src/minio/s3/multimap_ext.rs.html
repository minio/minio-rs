16use crate::s3::types::VersionId;
17use crate::s3::utils::url_encode;
18use std::borrow::Cow;
19use std::collections::BTreeMap;
20
21/// Multimap for string key and string value
22pub type Multimap = multimap::MultiMap<String, String>;
23
24/// Collapses multiple spaces into a single space (avoids regex overhead).
25///
26/// Returns `Cow::Borrowed` when no transformation is needed (common case),
27/// avoiding allocation for header values that don't contain consecutive spaces.
28#[inline]
29fn collapse_spaces(s: &str) -> Cow<'_, str> {
30    let trimmed = s.trim();
31    if !trimmed.contains("  ") {
32        return Cow::Borrowed(trimmed);
33    }
34    let mut result = String::with_capacity(trimmed.len());
35    let mut prev_space = false;
36    for c in trimmed.chars() {
37        if c == ' ' {
38            if !prev_space {
39                result.push(' ');
40                prev_space = true;
41            }
42        } else {
43            result.push(c);
44            prev_space = false;
45        }
46    }
47    Cow::Owned(result)
48}

50pub trait MultimapExt {
51    /// Adds a key-value pair to the multimap
52    fn add<K: Into<String>, V: Into<String>>(&mut self, key: K, value: V);
53
54    /// Adds a multimap to the current multimap
55    fn add_multimap(&mut self, other: Multimap);
56
57    fn add_version(&mut self, version: Option<VersionId>);
58
59    #[must_use]
60    fn take_version(self) -> Option<String>;
61
62    /// Converts multimap to HTTP query string
63    fn to_query_string(&self) -> String;
64
65    /// Converts multimap to canonical query string
66    fn get_canonical_query_string(&self) -> String;
67
68    /// Converts multimap to signed headers and canonical headers
69    fn get_canonical_headers(&self) -> (String, String);
70}
71
72impl MultimapExt for Multimap {
73    fn add<K: Into<String>, V: Into<String>>(&mut self, key: K, value: V) {
74        self.insert(key.into(), value.into());
75    }
76    fn add_multimap(&mut self, other: Multimap) {
77        for (key, values) in other.into_iter() {
78            self.insert_many(key, values);
79        }
80    }
81    fn add_version(&mut self, version: Option<VersionId>) {
82        if let Some(v) = version {
83            self.insert("versionId".into(), v.into_inner());
84        }
85    }
86    fn take_version(mut self) -> Option<String> {
87        self.remove("versionId").and_then(|mut v| v.pop())
88    }
89    fn to_query_string(&self) -> String {
90        let mut query = String::new();
91        for (key, values) in self.iter_all() {
92            for value in values {
93                if !query.is_empty() {
94                    query.push('&');
95                }
96                query.push_str(&url_encode(key));
97                query.push('=');
98                query.push_str(&url_encode(value));
99            }
100        }
101        query
102    }
103
104    fn get_canonical_query_string(&self) -> String {
105        // Use BTreeMap for automatic sorting (avoids explicit sort)
106        let mut sorted: BTreeMap<&str, Vec<&str>> = BTreeMap::new();
107        let mut total_len = 0usize;
108
109        for (key, values) in self.iter_all() {
110            for value in values {
111                // Pre-calculate total length to avoid reallocations.
112                // Most S3 query params are alphanumeric (uploadId, partNumber, versionId)
113                // so we use actual length + 20% buffer for occasional URL encoding.
114                total_len += key.len() + 1 + value.len() + 2; // key=value&
115            }
116            sorted
117                .entry(key.as_str())
118                .or_default()
119                .extend(values.iter().map(|s| s.as_str()));
120        }
121
122        // Add 20% buffer for URL encoding overhead
123        let mut query = String::with_capacity(total_len + total_len / 5);
124        for (key, values) in sorted {
125            for value in values {
126                if !query.is_empty() {
127                    query.push('&');
128                }
129                query.push_str(&url_encode(key));
130                query.push('=');
131                query.push_str(&url_encode(value));
132            }
133        }
134
135        query
136    }
137
138    fn get_canonical_headers(&self) -> (String, String) {
139        // Use BTreeMap for automatic sorting (avoids explicit sort)
140        let mut btmap: BTreeMap<String, String> = BTreeMap::new();
141
142        // Pre-calculate sizes for better allocation
143        let mut key_bytes = 0usize;
144        let mut value_bytes = 0usize;
145
146        for (k, values) in self.iter_all() {
147            let key = k.to_lowercase();
148            if key == "authorization" || key == "user-agent" {
149                continue;
150            }
151
152            // Sort values in place if needed
153            let mut vs: Vec<&String> = values.iter().collect();
154            vs.sort();
155
156            let mut value =
157                String::with_capacity(vs.iter().map(|v| v.len()).sum::<usize>() + vs.len());
158            for v in vs {
159                if !value.is_empty() {
160                    value.push(',');
161                }
162                value.push_str(&collapse_spaces(v));
163            }
164
165            key_bytes += key.len();
166            value_bytes += value.len();
167            btmap.insert(key, value);
168        }
169
170        // Pre-allocate output strings
171        let header_count = btmap.len();
172        let mut signed_headers = String::with_capacity(key_bytes + header_count);
173        let mut canonical_headers =
174            String::with_capacity(key_bytes + value_bytes + header_count * 2);
175
176        let mut add_delim = false;
177        for (key, value) in &btmap {
178            if add_delim {
179                signed_headers.push(';');
180                canonical_headers.push('\n');
181            }
182
183            signed_headers.push_str(key);
184
185            canonical_headers.push_str(key);
186            canonical_headers.push(':');
187            canonical_headers.push_str(value);
188
189            add_delim = true;
190        }
191
192        (signed_headers, canonical_headers)
193    }
194}

200    #[test]
201    fn test_collapse_spaces_no_consecutive_spaces() {
202        // Should return Cow::Borrowed (no allocation)
203        let result = collapse_spaces("hello world");
204        assert_eq!(result, "hello world");
205        assert!(matches!(result, Cow::Borrowed(_)));
206    }
207
208    #[test]
209    fn test_collapse_spaces_with_consecutive_spaces() {
210        // Should return Cow::Owned with spaces collapsed
211        let result = collapse_spaces("hello  world");
212        assert_eq!(result, "hello world");
213        assert!(matches!(result, Cow::Owned(_)));
214
215        let result = collapse_spaces("hello   world");
216        assert_eq!(result, "hello world");
217
218        let result = collapse_spaces("a  b  c  d");
219        assert_eq!(result, "a b c d");
220    }
