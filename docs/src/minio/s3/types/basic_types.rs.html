24#[derive(Clone, Debug)]
25/// Contains information of an item of [list_objects()](crate::s3::client::MinioClient::list_objects) API
26pub struct ListEntry {
27    pub name: String,
28    pub last_modified: Option<UtcTime>,
29    pub etag: Option<String>, // except DeleteMarker
30    pub owner_id: Option<String>,
31    pub owner_name: Option<String>,
32    pub size: Option<u64>, // except DeleteMarker
33    pub storage_class: Option<String>,
34    pub is_latest: bool,            // except ListObjects V1/V2
35    pub version_id: Option<String>, // except ListObjects V1/V2
36    pub user_metadata: Option<HashMap<String, String>>,
37    pub user_tags: Option<HashMap<String, String>>,
38    pub is_prefix: bool,
39    pub is_delete_marker: bool,
40    pub encoding_type: Option<String>,
41}

50#[derive(Clone, Debug)]
51/// Contains part number and etag of multipart upload
52pub struct Part {
53    pub number: u16,
54    pub etag: ETag,
55}
56
57/// Contains part information for multipart uploads including optional checksum.
58///
59/// Only one checksum algorithm is active per upload, so the checksum is stored
60/// as an optional tuple of (algorithm, base64-encoded value).
61#[derive(Clone, Debug)]
62pub struct PartInfo {
63    pub number: u16,
64    pub etag: ETag,
65    pub size: u64,
66    /// Optional checksum for this part: (algorithm, base64-encoded value)
67    pub checksum: Option<(ChecksumAlgorithm, String)>,
68}
69
70impl PartInfo {
71    /// Creates a new PartInfo.
72    pub fn new(
73        number: u16,
74        etag: ETag,
75        size: u64,
76        checksum: Option<(ChecksumAlgorithm, String)>,
77    ) -> Self {
78        Self {
79            number,
80            etag,
81            size,
82            checksum,
83        }
84    }
85}

87#[derive(PartialEq, Clone, Debug)]
88/// Contains retention mode information
89pub enum RetentionMode {
90    GOVERNANCE,
91    COMPLIANCE,
92}
93
94impl RetentionMode {
95    pub fn parse(s: &str) -> Result<RetentionMode, ValidationErr> {
96        if s.eq_ignore_ascii_case("GOVERNANCE") {
97            Ok(RetentionMode::GOVERNANCE)
98        } else if s.eq_ignore_ascii_case("COMPLIANCE") {
99            Ok(RetentionMode::COMPLIANCE)
100        } else {
101            Err(ValidationErr::InvalidRetentionMode(s.to_string()))
102        }
103    }
104}
105
106impl fmt::Display for RetentionMode {
107    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
108        match self {
109            RetentionMode::GOVERNANCE => write!(f, "GOVERNANCE"),
110            RetentionMode::COMPLIANCE => write!(f, "COMPLIANCE"),
111        }
112    }
113}

115#[derive(Clone, Debug)]
116/// Contains retention mode and retain until date
117pub struct Retention {
118    pub mode: RetentionMode,
119    pub retain_until_date: UtcTime,
120}
121
122/// Parses 'legal hold' string value
123pub fn parse_legal_hold(s: &str) -> Result<bool, ValidationErr> {
124    if s.eq_ignore_ascii_case("ON") {
125        Ok(true)
126    } else if s.eq_ignore_ascii_case("OFF") {
127        Ok(false)
128    } else {
129        Err(ValidationErr::InvalidLegalHold(s.to_string()))
130    }
131}
