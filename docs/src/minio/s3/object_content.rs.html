31#[derive(Debug, Clone, PartialEq, Eq, Copy, Default)]
32pub enum Size {
33    Known(u64),
34    #[default]
35    Unknown,
36}
37
38impl Size {
39    /// Returns `true` if the size is known and `false` otherwise.
40    pub fn is_known(&self) -> bool {
41        matches!(self, Size::Known(_))
42    }
43
44    /// Returns `true` if the size is unknown and `false` otherwise.
45    pub fn is_unknown(&self) -> bool {
46        matches!(self, Size::Unknown)
47    }
48
49    /// Returns the size if known, otherwise returns `None`.
50    pub fn value(&self) -> Option<u64> {
51        match self {
52            Size::Known(v) => Some(*v),
53            Size::Unknown => None,
54        }
55    }
56}
57
58impl From<Option<u64>> for Size {
59    fn from(value: Option<u64>) -> Self {
60        match value {
61            Some(v) => Size::Known(v),
62            None => Size::Unknown,
63        }
64    }
65}
66
67impl From<u64> for Size {
68    fn from(value: u64) -> Self {
69        Size::Known(value)
70    }
71}
72
73#[cfg(test)]
74impl Arbitrary for Size {
75    fn arbitrary(g: &mut quickcheck::Gen) -> Self {
76        if bool::arbitrary(g) {
77            Size::Known(u64::arbitrary(g))
78        } else {
79            Size::Unknown
80        }
81    }
82}

85/// Object content that can be uploaded or downloaded.
86///
87/// Can be constructed from a stream of `Bytes`, a file path, or a `Bytes` object.
88pub struct ObjectContent(ObjectContentInner);
89
90enum ObjectContentInner {
91    Stream(Pin<Box<dyn Stream<Item = IoResult<Bytes>> + Send>>, Size),
92    FilePath(PathBuf),
93    Bytes(SegmentedBytes),
94}
95
96impl From<Bytes> for ObjectContent {
97    fn from(value: Bytes) -> Self {
98        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::from(value)))
99    }
100}
101
102impl From<String> for ObjectContent {
103    fn from(value: String) -> Self {
104        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::from(
105            Bytes::from(value),
106        )))
107    }
108}
109
110impl From<Vec<u8>> for ObjectContent {
111    fn from(value: Vec<u8>) -> Self {
112        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::from(
113            Bytes::from(value),
114        )))
115    }
116}
117
118impl From<&'static [u8]> for ObjectContent {
119    fn from(value: &'static [u8]) -> Self {
120        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::from(
121            Bytes::from(value),
122        )))
123    }
124}
125
126impl From<&'static str> for ObjectContent {
127    fn from(value: &'static str) -> Self {
128        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::from(
129            Bytes::from(value),
130        )))
131    }
132}
133
134impl From<&Path> for ObjectContent {
135    fn from(value: &Path) -> Self {
136        ObjectContent(ObjectContentInner::FilePath(value.to_path_buf()))
137    }
138}
139
140impl Default for ObjectContent {
141    fn default() -> Self {
142        ObjectContent(ObjectContentInner::Bytes(SegmentedBytes::new()))
143    }
144}
145
146impl ObjectContent {
147    /// Create a new `ObjectContent` from a stream of `Bytes`.
148    pub fn new_from_stream(
149        r: impl Stream<Item = IoResult<Bytes>> + Send + 'static,
150        size: impl Into<Size>,
151    ) -> Self {
152        let r = Box::pin(r);
153        ObjectContent(ObjectContentInner::Stream(r, size.into()))
154    }
155
156    pub async fn to_stream(
157        self,
158    ) -> IoResult<(Pin<Box<dyn Stream<Item = IoResult<Bytes>> + Send>>, Size)> {
159        match self.0 {
160            ObjectContentInner::Stream(r, size) => Ok((r, size)),
161
162            ObjectContentInner::FilePath(path) => {
163                let mut file = async_std::fs::File::open(&path).await?;
164                let metadata = file.metadata().await?;
165                let size = metadata.len();
166
167                // Define a stream that reads the file in chunks
168                let stream = async_stream::try_stream! {
169                    let mut buf = vec![0u8; 8192];
170                    loop {
171                        let n = file.read(&mut buf).await?;
172                        if n == 0 {
173                            break;
174                        }
175                        yield Bytes::copy_from_slice(&buf[..n]);
176                    }
177                };
178
179                Ok((Box::pin(stream), Some(size).into()))
180            }
181
182            ObjectContentInner::Bytes(sb) => {
183                let k = sb.len();
184                let r = Box::pin(stream::iter(sb.into_iter().map(Ok)));
185                Ok((r, Some(k as u64).into()))
186            }
187        }
188    }
189
190    #[allow(clippy::wrong_self_convention)]
191    pub(crate) async fn to_content_stream(self) -> IoResult<ContentStream> {
192        let (r, size) = self.to_stream().await?;
193        Ok(ContentStream::new(r, size))
194    }
195
196    /// Load the content into memory and return a `SegmentedBytes` object.
197    pub async fn to_segmented_bytes(self) -> IoResult<SegmentedBytes> {
198        let mut segmented_bytes = SegmentedBytes::new();
199        let (mut r, _) = self.to_stream().await?;
200        while let Some(bytes) = r.next().await {
201            let bytes = bytes?;
202            if bytes.is_empty() {
203                break;
204            }
205            segmented_bytes.append(bytes);
206        }
207        Ok(segmented_bytes)
208    }
209
210    /// Write the content to a file. This function will return the total number
211    /// of bytes written to the file. It first writes the content to a temporary
212    /// file and then renames the temporary file to the final file path. The
213    /// temporary file will be located in the same directory as the final file
214    /// path.
215    ///
216    /// If the file already exists, it will be replaced. If the parent directory
217    /// does not exist, an attempt to create it will be made.
218    pub async fn to_file(self, file_path: &Path) -> IoResult<u64> {
219        if file_path.is_dir() {
220            return Err(std::io::Error::other("path is a directory"));
221        }
222        let parent_dir = file_path.parent().ok_or(std::io::Error::other(format!(
223            "path {file_path:?} does not have a parent directory"
224        )))?;
225        if !parent_dir.is_dir() {
226            async_std::fs::create_dir_all(parent_dir).await?;
227        }
228        let file_name = file_path.file_name().ok_or(std::io::Error::other(
229            "could not get filename-component of path",
230        ))?;
231        let mut tmp_file_name = file_name.to_os_string();
232        tmp_file_name.push(format!("_{}", Uuid::new_v4().to_string().replace('-', "_")));
233        let tmp_file_path = parent_dir.join(tmp_file_name);
234
235        let mut total_bytes_written = 0;
236        let mut fp = async_std::fs::OpenOptions::new()
237            .write(true)
238            .create(true) // Ensures that the file will be created if it does not already exist
239            .truncate(true) // Clears the contents (truncates the file size to 0) before writing
240            .open(&tmp_file_path)
241            .await?;
242        let (mut r, _) = self.to_stream().await?;
243        while let Some(bytes) = r.next().await {
244            let bytes = bytes?;
245            if bytes.is_empty() {
246                break;
247            }
248            total_bytes_written += bytes.len() as u64;
249            fp.write_all(&bytes).await?;
250        }
251        fp.flush().await?;
252        fs::rename(&tmp_file_path, file_path)?;
253        Ok(total_bytes_written)
254    }
255}

257pub struct ContentStream {
258    r: Pin<Box<dyn Stream<Item = IoResult<Bytes>> + Send>>,
259    extra: Option<Bytes>,
260    size: Size,
261}
262
263impl Default for ContentStream {
264    fn default() -> Self {
265        ContentStream::empty()
266    }
267}
268
269impl ContentStream {
270    pub fn new(
271        r: impl Stream<Item = IoResult<Bytes>> + Send + 'static,
272        size: impl Into<Size>,
273    ) -> Self {
274        let r = Box::pin(r);
275        Self {
276            r,
277            extra: None,
278            size: size.into(),
279        }
280    }
281
282    pub fn empty() -> Self {
283        Self {
284            r: Box::pin(stream::iter(vec![])),
285            extra: None,
286            size: Some(0).into(),
287        }
288    }
289
290    pub fn get_size(&self) -> Size {
291        self.size
292    }
293
294    // Read as many bytes as possible up to `n` and return a `SegmentedBytes`
295    // object.
296    pub async fn read_upto(&mut self, n: usize) -> IoResult<SegmentedBytes> {
297        let mut segmented_bytes = SegmentedBytes::new();
298        let mut remaining = n;
299        if let Some(extra) = self.extra.take() {
300            let len = extra.len();
301            if len <= remaining {
302                segmented_bytes.append(extra);
303                remaining -= len;
304            } else {
305                segmented_bytes.append(extra.slice(0..remaining));
306                self.extra = Some(extra.slice(remaining..));
307                return Ok(segmented_bytes);
308            }
309        }
310        while remaining > 0 {
311            let bytes = self.r.next().await;
312            if bytes.is_none() {
313                break;
314            }
315            let bytes = bytes.unwrap()?;
316            let len = bytes.len();
317            if len == 0 {
318                break;
319            }
320            if len <= remaining {
321                segmented_bytes.append(bytes);
322                remaining -= len;
323            } else {
324                segmented_bytes.append(bytes.slice(0..remaining));
325                self.extra = Some(bytes.slice(remaining..));
326                break;
327            }
328        }
329        Ok(segmented_bytes)
330    }
