28/// Argument builder for the [`HeadObject`](https://docs.aws.amazon.com/AmazonS3/latest/API/API_HeadObject.html) S3 API operation.
29///
30/// This struct constructs the parameters required for the [`Client::stat_object`](crate::s3::client::MinioClient::stat_object) method.
31///
32/// # HTTP Method
33///
34/// This operation uses the HTTP HEAD method, which retrieves object metadata
35/// without transferring the object body. This is more efficient than GET when
36/// you only need metadata (size, ETag, Content-Type, Last-Modified, etc.).
37#[derive(Debug, Clone, TypedBuilder)]
38pub struct StatObject {
39    #[builder(!default)] // force required
40    client: MinioClient,
41
42    #[builder(default, setter(into))]
43    extra_headers: Option<Multimap>,
44    #[builder(default, setter(into))]
45    extra_query_params: Option<Multimap>,
46    #[builder(setter(into), !default)]
47    bucket: BucketName,
48    #[builder(setter(into), !default)]
49    object: ObjectKey,
50
51    #[builder(default, setter(into))]
52    version_id: Option<VersionId>,
53    #[builder(default, setter(into))]
54    region: Option<Region>,
55    #[builder(default, setter(into))]
56    ssec: Option<SseCustomerKey>,
57
58    // Conditionals
59    #[builder(default, setter(into))]
60    match_etag: Option<String>,
61    #[builder(default, setter(into))]
62    not_match_etag: Option<String>,
63    #[builder(default, setter(into))]
64    modified_since: Option<UtcTime>,
65    #[builder(default, setter(into))]
66    unmodified_since: Option<UtcTime>,
67}
68

91#[async_trait]
92impl ToS3Request for StatObject {
93    fn to_s3request(self) -> Result<S3Request, ValidationErr> {
94        check_ssec(&self.ssec, &self.client)?;
95
96        let mut headers: Multimap = self.extra_headers.unwrap_or_default();
97        {
98            if let Some(v) = self.match_etag {
99                headers.add(IF_MATCH, v);
100            }
101            if let Some(v) = self.not_match_etag {
102                headers.add(IF_NONE_MATCH, v);
103            }
104            if let Some(v) = self.modified_since {
105                headers.add(IF_MODIFIED_SINCE, to_http_header_value(v));
106            }
107            if let Some(v) = self.unmodified_since {
108                headers.add(IF_UNMODIFIED_SINCE, to_http_header_value(v));
109            }
110            if let Some(v) = self.ssec {
111                headers.add_multimap(v.headers());
112            }
113        }
114
115        let mut query_params: Multimap = self.extra_query_params.unwrap_or_default();
116        query_params.add_version(self.version_id);
117
118        Ok(S3Request::builder()
119            .client(self.client)
120            .method(Method::HEAD)
121            .region(self.region)
122            .bucket(self.bucket)
123            .object(self.object)
124            .query_params(query_params)
125            .headers(headers)
126            .build())
127    }
