16//! MinIO Extension API for S3 Buckets: ListenBucketNotification
17
18use crate::s3::builders::{ListenBucketNotification, ListenBucketNotificationBldr};
19use crate::s3::client::MinioClient;
20use crate::s3::error::ValidationErr;
21use crate::s3::types::BucketName;
22
23impl MinioClient {
24    /// Creates a [`ListenBucketNotification`] request builder.
25    ///
26    /// To execute the request, call [`ListenBucketNotification::send()`](crate::s3::types::S3Api::send),
27    /// which returns a tuple of [`ListenBucketNotificationResponse`](crate::s3::response::ListenBucketNotificationResponse) and a
28    /// stream of [`NotificationRecords`](crate::s3::types::NotificationRecords). The former contains the HTTP headers
29    /// returned by the server and the latter is a stream of notification
30    /// records. In normal operation (when there are no errors), the stream
31    /// never ends.
32    ///
33    /// # MinIO Extensions
34    ///
35    /// This function is only available in MinIO and not part of the AWS S3 API.
36    ///
37    /// # Example
38    ///
39    /// ```no_run
40    /// use minio::s3::MinioClient;
41    /// use minio::s3::creds::StaticProvider;
42    /// use minio::s3::http::BaseUrl;
43    /// use minio::s3::types::{NotificationRecord, NotificationRecords, S3Api};
44    /// use futures_util::StreamExt;
45    ///
46    /// #[tokio::main]
47    /// async fn main() {
48    ///     let base_url = "http://localhost:9000/".parse::<BaseUrl>().unwrap();
49    ///     let static_provider = StaticProvider::new("minioadmin", "minioadmin", None);
50    ///     let client = MinioClient::new(base_url, Some(static_provider), None, None).unwrap();
51    ///     let (_resp, mut event_stream) = client
52    ///         .listen_bucket_notification("bucket-name").unwrap()
53    ///         .build().send().await.unwrap();
54    ///
55    ///     while let Some(event) = event_stream.next().await {
56    ///         let event: NotificationRecords = event.unwrap();
57    ///         let record: Option<&NotificationRecord> = event.records.first();
58    ///         println!("received a notification record {:#?}", record);
59    ///     }
60    /// }
61    /// ```
62    pub fn listen_bucket_notification<B>(
63        &self,
64        bucket: B,
65    ) -> Result<ListenBucketNotificationBldr, ValidationErr>
66    where
67        B: TryInto<BucketName>,
68        B::Error: Into<ValidationErr>,
69    {
70        Ok(ListenBucketNotification::builder()
71            .client(self.clone())
72            .bucket(bucket.try_into().map_err(Into::into)?))
73    }
74}
