33impl MinioClient {
34    /// Creates a [`DeleteBucket`] request builder.
35    ///
36    /// To execute the request, call [`DeleteBucket::send()`](crate::s3::types::S3Api::send),
37    /// which returns a [`Result`] containing a [`DeleteBucketResponse`].
38    ///
39    /// # Example
40    ///
41    /// ```no_run
42    /// use minio::s3::MinioClient;
43    /// use minio::s3::creds::StaticProvider;
44    /// use minio::s3::http::BaseUrl;
45    /// use minio::s3::response::DeleteBucketResponse;
46    /// use minio::s3::types::S3Api;
47    /// use minio::s3::response_traits::{HasBucket, HasRegion};
48    ///
49    /// #[tokio::main]
50    /// async fn main() {
51    ///     let base_url = "http://localhost:9000/".parse::<BaseUrl>().unwrap();
52    ///     let static_provider = StaticProvider::new("minioadmin", "minioadmin", None);
53    ///     let client = MinioClient::new(base_url, Some(static_provider), None, None).unwrap();
54    ///     let resp: DeleteBucketResponse = client
55    ///         .delete_bucket("bucket-name")
56    ///         .unwrap().build().send().await.unwrap();
57    ///     println!("bucket '{}' in region '{}' is removed", resp.bucket().unwrap(), resp.region());
58    /// }
59    /// ```
60    pub fn delete_bucket<B>(&self, bucket: B) -> Result<DeleteBucketBldr, ValidationErr>
61    where
62        B: TryInto<BucketName>,
63        B::Error: Into<ValidationErr>,
64    {
65        Ok(DeleteBucket::builder()
66            .client(self.clone())
67            .bucket(bucket.try_into().map_err(Into::into)?))
68    }

70    /// Deletes a bucket and also deletes non-empty buckets by first removing all objects before
71    /// deleting the bucket. Bypasses governance mode and legal hold.
72    pub async fn delete_and_purge_bucket<B>(&self, bucket: B) -> Result<DeleteBucketResponse, Error>
73    where
74        B: TryInto<BucketName>,
75        B::Error: Into<ValidationErr>,
76    {
77        let bucket: BucketName = bucket.try_into().map_err(Into::into)?;
78        let resp: BucketExistsResponse = self.bucket_exists(&bucket)?.build().send().await?;
79        if !resp.exists {
80            // if the bucket does not exist, we can return early
81            let dummy: S3Request = S3Request::builder()
82                .client(self.clone())
83                .method(Method::DELETE)
84                .bucket(bucket)
85                .headers(MultiMap::default())
86                .build();
87
88            return Ok(DeleteBucketResponse {
89                request: dummy, //TODO consider how to handle this
90                body: Bytes::new(),
91                headers: Default::default(),
92            });
93        }
94
95        let is_express = self.is_minio_express().await;
96
97        let mut stream = self
98            .list_objects(&bucket)?
99            .include_versions(!is_express)
100            .recursive(true)
101            .build()
102            .to_stream()
103            .await;
104
105        if is_express {
106            while let Some(items) = stream.next().await {
107                let object_names = items?.contents.into_iter().map(ObjectToDelete::from);
108                let mut resp = self
109                    .delete_objects_streaming(&bucket, object_names)?
110                    .bypass_governance_mode(false) // Express does not support governance mode
111                    .to_stream()
112                    .await;
113
114                while let Some(item) = resp.next().await {
115                    let _resp: DeleteObjectsResponse = item?;
116                }
117            }
118        } else {
119            while let Some(items) = stream.next().await {
120                let object_names = items?.contents.into_iter().map(ObjectToDelete::from);
121                let mut resp = self
122                    .delete_objects_streaming(&bucket, object_names)?
123                    .bypass_governance_mode(true)
124                    .to_stream()
125                    .await;
126
127                while let Some(item) = resp.next().await {
128                    let resp: DeleteObjectsResponse = item?;
129                    for obj in resp.result()?.into_iter() {
130                        match obj {
131                            DeleteResult::Deleted(_) => {}
132                            DeleteResult::Error(v) => {
133                                // the object is not deleted. try to disable legal hold and try again.
134                                let object_key =
135                                    ObjectKey::try_from(v.object_name.as_str()).unwrap();
136
137                                let result = match &v.version_id {
138                                    Some(vid) => {
139                                        self.put_object_legal_hold(&bucket, &object_key, false)?
140                                            .version_id(VersionId::try_from(vid.as_str()).unwrap())
141                                            .build()
142                                            .send()
143                                            .await
144                                    }
145                                    None => {
146                                        self.put_object_legal_hold(&bucket, &object_key, false)?
147                                            .build()
148                                            .send()
149                                            .await
150                                    }
151                                };
152
153                                let _resp: PutObjectLegalHoldResponse = result?;
154
155                                let _resp: DeleteObjectResponse = DeleteObject::builder()
156                                    .client(self.clone())
157                                    .bucket(&bucket)
158                                    .object(v)
159                                    .bypass_governance_mode(true)
160                                    .build()
161                                    .send()
162                                    .await?;
163                            }
164                        }
165                    }
166                }
167            }
168        }
169
170        let request: DeleteBucket = self.delete_bucket(&bucket)?.build();
171        match request.send().await {
172            Ok(resp) => Ok(resp),
173            Err(Error::S3Server(S3Error(mut e))) => {
174                if matches!(e.code(), MinioErrorCode::NoSuchBucket) {
175                    let dummy: S3Request = S3Request::builder()
176                        .client(self.clone())
177                        .method(Method::DELETE)
178                        .bucket(bucket)
179                        .headers(MultiMap::default())
180                        .build();
181
182                    Ok(DeleteBucketResponse {
183                        request: dummy, //TODO consider how to handle this
184                        body: Bytes::new(),
185                        headers: e.take_headers(),
186                    })
187                } else if matches!(e.code(), MinioErrorCode::BucketNotEmpty) {
188                    // for convenience, add the first 5 documents that were are still in the bucket
189                    // to the error message
190                    let mut stream = self
191                        .list_objects(&bucket)?
192                        .include_versions(!is_express)
193                        .recursive(true)
194                        .build()
195                        .to_stream()
196                        .await;
197
198                    let mut objs = Vec::new();
199                    while let Some(items_result) = stream.next().await {
200                        if let Ok(items) = items_result {
201                            objs.extend(items.contents);
202                            if objs.len() >= 5 {
203                                break;
204                            }
205                        }
206                        // else: silently ignore the error and keep looping
207                    }
208
209                    let new_msg = match e.message() {
210                        None => format!("found content: {objs:?}"),
211                        Some(msg) => format!("{msg}, found content: {objs:?}"),
212                    };
213                    e.set_message(new_msg);
214                    Err(Error::S3Server(S3Error(e)))
215                } else if e
216                    .message()
217                    .as_ref()
218                    .map(|msg| msg.contains("Use DeleteWarehouse API"))
219                    .unwrap_or(false)
220                {
221                    // This is a warehouse bucket - provide helpful guidance
222                    let original_msg = e.message().clone().unwrap_or_default();
223                    let new_msg = format!(
224                        "Cannot delete warehouse bucket '{}' using DeleteBucket API. \
225                         Warehouse buckets must be deleted using the DeleteWarehouse S3 Tables API. \
226                         Original error: {}",
227                        bucket, original_msg
228                    );
229                    e.set_code(MinioErrorCode::WarehouseBucketOperationNotSupported);
230                    e.set_message(new_msg);
231                    Err(Error::S3Server(S3Error(e)))
232                } else {
233                    Err(Error::S3Server(S3Error(e)))
234                }
235            }
236            Err(e) => Err(e),
237        }
238    }
