19impl MinioClient {
20    /// Creates a [`GetPresignedPolicyFormData`] request builder.
21    ///
22    /// To execute the request, call [`GetPresignedPolicyFormData::send()`](crate::s3::types::S3Api::send),
23    /// which returns a `HashMap<String, String>` with the presigned policy.
24    ///
25    /// # Example
26    ///
27    /// ```no_run
28    /// use http::Method;
29    /// use std::collections::HashMap;
30    /// use chrono::{DateTime, Utc};
31    /// use minio::s3::MinioClient;
32    /// use minio::s3::creds::StaticProvider;
33    /// use minio::s3::http::BaseUrl;
34    /// use minio::s3::types::{BucketName, S3Api};
35    /// use minio::s3::builders::PostPolicy;
36    /// use minio::s3::utils::utc_now;
37    ///
38    /// #[tokio::main]
39    /// async fn main() {
40    ///     let base_url = "http://localhost:9000/".parse::<BaseUrl>().unwrap();
41    ///     let static_provider = StaticProvider::new("minioadmin", "minioadmin", None);
42    ///     let client = MinioClient::new(base_url, Some(static_provider), None, None).unwrap();
43    ///
44    ///     let expiration: DateTime<Utc> = utc_now() + chrono::Duration::days(5);
45    ///     let mut policy = PostPolicy::new(BucketName::new("bucket-name").unwrap(), expiration).unwrap();
46    ///     policy.add_equals_condition("key", "object-name").unwrap();
47    ///     policy.add_content_length_range_condition(1024 * 1024, 4 * 1024 * 1024).unwrap();
48    ///
49    ///     let resp: HashMap<String, String> = client
50    ///         .get_presigned_post_form_data(policy)
51    ///         .build().send().await.unwrap();
52    ///     println!("presigned post form data: '{:?}'", resp);
53    /// }
54    /// ```
55    pub fn get_presigned_post_form_data(
56        &self,
57        policy: PostPolicy,
58    ) -> GetPresignedPolicyFormDataBldr {
59        GetPresignedPolicyFormData::builder()
60            .client(self.clone())
61            .policy(policy)
62    }
