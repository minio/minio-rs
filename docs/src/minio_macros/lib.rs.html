17use darling::ast::NestedMeta;
18use darling::{Error, FromMeta};
19use syn::ItemFn;
20extern crate proc_macro;

22/// A proc macro attribute for writing MinIO tests.
23///
24/// This macro extends the `#[tokio::test]` attribute to provide additional functionality for
25/// testing MinIO operations. The macro takes care of setting up and tearing down the test
26/// environment, it automatically creates a bucket for the test if needed and cleans it up after
27/// the test is done.
28///
29/// By default, it requires the test function to have two parameters:
30///
31/// - `ctx: TestContext` - The test context which will give you access to a minio-client.
32/// - `bucket_name: BucketName` - The typed bucket name to be used in the test.
33///
34/// ```no_run
35/// use minio_common::test_context::TestContext;
36/// use minio::s3::types::BucketName;
37/// #[minio_macros::test]
38/// async fn my_test(ctx: TestContext, bucket_name: BucketName) {
39///    // Your test code here
40/// }
41/// ```
42///
43/// If the `no_bucket` argument is provided, the test function must have only one parameter:
44///
45/// - `ctx: TestContext` - The test context which will give you access to a minio-client.
46///
47/// ```no_run
48/// use minio_common::test_context::TestContext;
49/// #[minio_macros::test(no_bucket)]
50/// async fn my_test(ctx: TestContext) {
51///    // Your test code here
52/// }
53///```
54/// The macro also supports additional arguments:
55///
56/// - `flavor`: Specifies the flavor of the Tokio test (e.g., "multi_thread").
57/// - `worker_threads`: Specifies the number of worker threads for the Tokio test.
58/// - `bucket_name`: Specifies the name of the bucket to be used in the test. If not provided, a random bucket name will be generated.
59/// - `skip_if_express`: If set, the test will be skipped if the MinIO server is running in Express mode.
60/// - `object_lock`: If set, the test bucket is created with `.object_lock(true)`
61/// - `no_cleanup`: If set, the test bucket is not deleted after the test is run.
62///
63/// ```no_run
64/// use minio_common::test_context::TestContext;
65/// #[minio_macros::test(skip_if_express)]
66/// async fn my_test(ctx: TestContext) {
67///    // this test will not run if the MinIO server is running in Express mode
68/// }
69/// ```
70/// - `skip_if_not_express`: If set, the test will be skipped if the MinIO server is NOT running in Express mode.
71/// ```no_run
72/// use minio_common::test_context::TestContext;
73/// #[minio_macros::test(skip_if_not_express)]
74/// async fn my_test(ctx: TestContext) {
75///    // this test will not run if the MinIO server is NOT running in Express mode
76/// }
77/// ```
78/// - `ignore`: Mark the test as ignored (skipped by default). Run with `cargo test -- --ignored` to include.
79/// ```no_run
80/// use minio_common::test_context::TestContext;
81/// use minio::s3::types::BucketName;
82/// #[minio_macros::test(ignore = "Requires newer server version")]
83/// async fn my_test(ctx: TestContext, bucket_name: BucketName) {
84///    // this test is skipped by default
85/// }
86/// ```
87#[proc_macro_attribute]
88pub fn test(
89    args: proc_macro::TokenStream,
90    input: proc_macro::TokenStream,
91) -> proc_macro::TokenStream {
92    // Parse the function
93    let input_fn = match syn::parse::<ItemFn>(input.clone()) {
94        Ok(input_fn) => input_fn,
95        Err(err) => return err.to_compile_error().into(),
96    };
97
98    // Parse the macro arguments
99    let attr_args = match NestedMeta::parse_meta_list(args.into()) {
100        Ok(v) => v,
101        Err(e) => return Error::from(e).write_errors().into(),
102    };
103
104    let args = match test_attr::MacroArgs::from_list(&attr_args) {
105        Ok(v) => v,
106        Err(e) => return e.write_errors().into(),
107    };
108
109    // Validate the function arguments
110    if let Err(err) = args.validate(&input_fn) {
111        return err;
112    }
113
114    // Expand the macro
115    match test_attr::expand_test_macro(args, input_fn) {
116        Ok(expanded) => expanded.into(),
117        Err(err) => err,
118    }
119}
