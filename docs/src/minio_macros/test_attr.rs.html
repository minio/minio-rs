25#[derive(Debug, FromMeta)]
26pub(crate) struct MacroArgs {
27    flavor: Option<String>,
28    worker_threads: Option<usize>,
29    bucket_name: Option<String>,
30    skip_if_express: darling::util::Flag,
31    skip_if_not_express: darling::util::Flag,
32    no_bucket: darling::util::Flag,
33    object_lock: darling::util::Flag,
34    no_cleanup: darling::util::Flag,
35    /// Mark test as ignored (skipped by default, run with `cargo test -- --ignored`)
36    ignore: Option<String>,
37}
38
39impl MacroArgs {
40    pub(crate) fn validate(&self, func: &ItemFn) -> Result<(), proc_macro::TokenStream> {
41        if self.no_bucket.is_present() && self.bucket_name.is_some() {
42            let error_msg = "The `no_bucket` argument cannot be used with `bucket_name`";
43            return Err(proc_macro::TokenStream::from(
44                Error::custom(error_msg)
45                    .with_span(&func.sig.span())
46                    .write_errors(),
47            ));
48        }
49
50        if self.no_bucket.is_present() && func.sig.inputs.len() != 1 {
51            let error_msg = "When using `no_bucket`, the test function must have exactly one argument: (ctx: TestContext)";
52            return Err(proc_macro::TokenStream::from(
53                Error::custom(error_msg)
54                    .with_span(&func.sig.inputs.span())
55                    .write_errors(),
56            ));
57        }
58
59        // Validate that the function has exactly two arguments: ctx and bucket_name
60        if (func.sig.inputs.len() != 2) && !self.no_bucket.is_present() {
61            let error_msg = "Minio test function must have exactly two arguments: (ctx: TestContext, bucket_name: BucketName)";
62            return Err(proc_macro::TokenStream::from(
63                Error::custom(error_msg)
64                    .with_span(&func.sig.inputs.span())
65                    .write_errors(),
66            ));
67        }
68
69        // Check the argument types
70        let mut iter = func.sig.inputs.iter();
71
72        // Check first argument (ctx: &mut TestContext)
73        if let Some(FnArg::Typed(pat_type)) = iter.next() {
74            let type_str = pat_type.ty.to_token_stream().to_string();
75            if !type_str.contains("TestContext") {
76                let error_msg = "The first argument must be of type TestContext";
77                return Err(proc_macro::TokenStream::from(
78                    Error::custom(error_msg)
79                        .with_span(&pat_type.span())
80                        .write_errors(),
81                ));
82            }
83        }
84
85        // Check the second argument (bucket_name: BucketName)
86        if !self.no_bucket.is_present()
87            && let Some(FnArg::Typed(pat_type)) = iter.next()
88        {
89            let type_str = pat_type.ty.to_token_stream().to_string();
90            if !type_str.contains("BucketName") {
91                let error_msg = "The second argument must be of type BucketName";
92                return Err(proc_macro::TokenStream::from(
93                    Error::custom(error_msg)
94                        .with_span(&pat_type.span())
95                        .write_errors(),
96                ));
97            }
98        }
99
100        Ok(())
101    }
102}

104/// Expands the test macro into the final TokenStream
105pub(crate) fn expand_test_macro(
106    args: MacroArgs,
107    mut func: ItemFn,
108) -> Result<TokenStream, proc_macro::TokenStream> {
109    let input_span = func.sig.paren_token.span.span();
110    let old_output = func.sig.output.clone();
111    let returns_result = returns_result_type(&old_output);
112    func.sig.output = ReturnType::Default;
113    let old_inps = func.sig.inputs.clone();
114    func.sig.inputs = Punctuated::default();
115    let sig = func.sig.clone().into_token_stream();
116
117    // Generate the tokio test attribute based on the provided arguments
118    let header = generate_tokio_test_header(&args, sig);
119
120    let test_function_block = func.block.clone().into_token_stream();
121
122    let inner_inputs = quote_spanned!(input_span=> #old_inps);
123    let inner_fn_name = create_inner_func_name(&func);
124    let inner_header =
125        quote_spanned!(func.sig.span()=> async fn #inner_fn_name(#inner_inputs) #old_output);
126
127    // Generate the skip logic for express mode if required
128    let maybe_skip_if_express = generate_express_skip_logic(&args, func.sig.span());
129
130    // Setup common prelude
131    let prelude = quote!(
132            use ::futures_util::FutureExt;
133            use ::std::panic::AssertUnwindSafe;
134            use ::minio::s3::types::S3Api;
135            use ::minio::s3::response_traits::HasBucket;
136
137            let ctx = ::minio_common::test_context::TestContext::new_from_env();
138    );
139
140    // Generate the outer function body based on whether a bucket is needed
141    let outer_body = if args.no_bucket.is_present() {
142        generate_no_bucket_body(
143            prelude,
144            maybe_skip_if_express,
145            inner_fn_name,
146            func.block.span(),
147            returns_result,
148        )
149    } else {
150        generate_with_bucket_body(
151            prelude,
152            maybe_skip_if_express,
153            inner_fn_name,
154            &args,
155            func.block.span(),
156            returns_result,
157        )
158    };
159
160    // Generate the inner function implementation
161    let inner_impl = quote_spanned!(func.span()=>
162        #inner_header
163        #test_function_block
164    );
165
166    // Combine all parts into the final output
167    let mut out = TokenStream::new();
168    out.extend(header);
169    out.extend(outer_body);
170    out.extend(inner_impl);
171
172    Ok(out)
173}

175fn generate_tokio_test_header(args: &MacroArgs, sig: TokenStream) -> TokenStream {
176    let flavor = args
177        .flavor
178        .as_ref()
179        .map(ToString::to_string)
180        .or(std::env::var("MINIO_TEST_TOKIO_RUNTIME_FLAVOR").ok());
181
182    // Generate #[ignore = "reason"] if specified
183    let ignore_attr = args
184        .ignore
185        .as_ref()
186        .map(|reason| quote!(#[ignore = #reason]));
187
188    let tokio_attr = match (flavor, args.worker_threads) {
189        (Some(flavor), None) => {
190            quote!(#[::tokio::test(flavor = #flavor)])
191        }
192        (None, Some(worker_threads)) => {
193            quote!(#[::tokio::test(worker_threads = #worker_threads)])
194        }
195        (None, None) => {
196            quote!(#[::tokio::test])
197        }
198        (Some(flavor), Some(worker_threads)) => {
199            quote!(#[::tokio::test(flavor = #flavor, worker_threads = #worker_threads)])
200        }
201    };
202
203    quote!(
204        #ignore_attr
205        #tokio_attr
206        #sig
207    )
208}

210fn generate_express_skip_logic(args: &MacroArgs, span: proc_macro2::Span) -> TokenStream {
211    if args.skip_if_express.is_present() {
212        quote_spanned!(span=>
213        if ctx.client.is_minio_express().await {
214            println!("Skipping test because it is running in MinIO Express mode");
215            return;
216        })
217    } else if args.skip_if_not_express.is_present() {
218        quote_spanned!(span=>
219        if !ctx.client.is_minio_express().await {
220            println!("Skipping test because it is NOT running in MinIO Express mode");
221            return;
222        })
223    } else {
224        TokenStream::new()
225    }
226}

228fn returns_result_type(output: &ReturnType) -> bool {
229    match output {
230        ReturnType::Default => false,
231        ReturnType::Type(_, ty) => {
232            let type_str = ty.to_token_stream().to_string();
233            type_str.starts_with("Result") || type_str.contains(":: Result")
234        }
235    }
236}

238fn generate_no_bucket_body(
239    prelude: TokenStream,
240    maybe_skip_if_express: TokenStream,
241    inner_fn_name: TokenStream,
242    span: proc_macro2::Span,
243    returns_result: bool,
244) -> TokenStream {
245    let inner_call = if returns_result {
246        quote_spanned!(span=>
247            if let Err(e) = #inner_fn_name(ctx).await {
248                panic!("Test failed with error: {:?}", e);
249            }
250        )
251    } else {
252        quote_spanned!(span=> #inner_fn_name(ctx).await;)
253    };
254    quote_spanned!(span=> {
255        #prelude
256        #maybe_skip_if_express
257        #inner_call
258    })
259}

261fn generate_with_bucket_body(
262    prelude: TokenStream,
263    maybe_skip_if_express: TokenStream,
264    inner_fn_name: TokenStream,
265    args: &MacroArgs,
266    span: proc_macro2::Span,
267    returns_result: bool,
268) -> TokenStream {
269    let bucket_name = args
270        .bucket_name
271        .as_ref()
272        .map(|b| b.to_token_stream())
273        .unwrap_or_else(|| {
274            let random_name = format!("test-bucket-{}", Uuid::new_v4());
275            proc_macro2::Literal::string(&random_name).into_token_stream()
276        });
277    let maybe_lock = if args.object_lock.is_present() {
278        quote! {
279            .object_lock(true)
280        }
281    } else {
282        TokenStream::new()
283    };
284    let (maybe_save_bucket, maybe_cleanup) = if args.no_cleanup.is_present() {
285        (quote! {}, quote! {})
286    } else {
287        (
288            quote! {
289                let bucket_name_for_cleanup = bucket_name.clone();
290            },
291            quote! {
292                ::minio_common::cleanup_guard::cleanup(client_clone, bucket_name_for_cleanup).await;
293            },
294        )
295    };
296    let inner_call = if returns_result {
297        quote_spanned!(span=>
298            let res = AssertUnwindSafe(async {
299                #inner_fn_name(ctx, bucket_name).await.map_err(|e| panic!("Test failed with error: {:?}", e))
300            }).catch_unwind().await;
301        )
302    } else {
303        quote_spanned!(span=>
304            let res = AssertUnwindSafe(#inner_fn_name(ctx, bucket_name)).catch_unwind().await;
305        )
306    };
307    quote_spanned!(span=> {
308        #prelude
309        #maybe_skip_if_express
310
311        let client_clone = ctx.client.clone();
312        let bucket_name_str = #bucket_name;
313        let bucket_name = ::minio::s3::types::BucketName::try_from(bucket_name_str).expect("Invalid bucket name");
314        // Try to create bucket, but continue if it already exists (for no_cleanup tests)
315        match client_clone.create_bucket(bucket_name.clone()).expect("Failed to create bucket builder")#maybe_lock.build().send().await {
316            Ok(resp) => {
317                assert_eq!(resp.bucket(), Some(&bucket_name));
318            }
319            Err(e) => {
320                // If bucket already exists, that's ok for no_cleanup tests
321                let err_str = format!("{:?}", e);
322                if !err_str.contains("BucketAlreadyOwnedByYou") && !err_str.contains("BucketAlreadyExists") {
323                    panic!("Failed to create bucket: {:?}", e);
324                }
325                // Otherwise continue - bucket already exists from previous run
326                eprintln!("Note: Reusing existing bucket {} from previous test run", bucket_name_str);
327            }
328        };
329        #maybe_save_bucket
330        #inner_call
331        #maybe_cleanup
332        if let Err(e) = res {
333            ::std::panic::resume_unwind(e);
334        }
335    })
336}

338fn create_inner_func_name(func: &ItemFn) -> TokenStream {
339    let inner_name = format!("{}_test_impl", func.sig.ident);
340    let ident = proc_macro2::Ident::new(&inner_name, func.sig.span());
341    quote! { #ident }
